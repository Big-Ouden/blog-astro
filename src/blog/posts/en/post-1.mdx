manage_sparse_checkout() {
  # V√©rification qu'on est dans un repo git
  if ! git rev-parse --is-inside-work-tree >/dev/null 2>&1; then
    echo -e "${RED}Erreur:${RESET} Cette fonction doit √™tre ex√©cut√©e dans un d√©p√¥t git."
    exit 1

  case "$1" in

        git sparse-checkout list | sed 's/^/  ‚úì /'

        echo "  Sparse-checkout non configur√© (tout le repo est synchronis√©)"
      fi

        exit 1

      if ! git sparse-checkout list >/dev/null 2>&1; then
        git sparse-checkout init --cone
      fi
      git sparse-checkout add "$2"
      git read-tree -m -u HEAD
      echo -e "${GREEN}‚úì '$2' ajout√© √† la synchronisation${RESET}"
      ;;
    remove)

        echo -e "${RED}Erreur:${RESET} Il faut sp√©cifier un nom de rapport."
        exit 1
      fi
      echo -e "${YELLOW}Suppression de '$2' de la synchronisation...${RESET}"
      current_patterns=$(git sparse-checkout list | grep -v "^$2$" || true)

        echo -e "${YELLOW}Attention: Plus aucun dossier ne sera synchronis√©. R√©initialisation compl√®te.${RESET}"
